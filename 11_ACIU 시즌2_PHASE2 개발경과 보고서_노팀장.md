```python
def get_korea_time():
    korea_tz = pytz.timezone('Asia/Seoul')
    now = datetime.now(korea_tz)
    return {
        'formatted': now.strftime('%Y년 %m월 %d일 %H:%M KST'),
        'iso': now.isoformat(),
        'timestamp': int(now.timestamp())
    }
```
